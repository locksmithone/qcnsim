183  // Creates a PDU and assigns it class member values and a unique ID obtained from simulatorGlobals.
184  // Notice that the base class member function "createInstance..." is used here instead of the derived override:
185  // the derived override schedules an event, and that is undesired here, since an event can only be scheduled "after" the PDU is created.
186  std::shared_ptr<Token> token = TrafficGenerator::createInstanceTrafficEvent(recordRoute);

189  }
190  //std::shared_ptr<ProtocolDataUnit> pdu(new ProtocolDataUnit(TrafficGenerator::createInstanceTrafficEvent(recordRoute), pduSize));
191  std::shared_ptr<ProtocolDataUnit> pdu(new ProtocolDataUnit(token, pduSize));
192  if (pdu != nullptr) {
193  scheduler.schedule(Event(generateExponentialVariate(), eventType, pdu));
194  }
195  return pdu;
