190  // Creates a PDU and assigns it class member values and a unique ID obtained from simulatorGlobals.
191  // Notice that the base class member function "createInstance..." is used here instead of the derived override:
192  // the derived override schedules an event, and that is undesired here, since an event can only be scheduled "after" the PDU is created.

195  return nullptr;
196  }
197  std::shared_ptr<ProtocolDataUnit> pdu(new ProtocolDataUnit(token, pduSize));

199  if (pdu != nullptr) {
200  scheduler.schedule(Event(generateWeibullVariate(), eventType, pdu));
