157 LinkReturnType Link::transmitPDU(EventType transmitEventType, EventType endTransmitEventType, std::shared_ptr<const ProtocolDataUnit> pdu) {
158  // Consistency check. If nodes in PDU's previous and next fields are not connected by this node, then refuse transmission and return error state.
159  if (pdu->previous != nodeA || pdu->next != nodeB) {
160  return LinkReturnType::ERROR_LINK_DOES_NOT_CONNECT_NODES;
161  }
162  double transmissionTime = pdu->getPduSize() * 8.0 / bandwidth; // Converts PDU size to bits and calculates transmission time based on bandwidth.
163  // Request service for PDU at Facility (transmission server).
164  FacilityReturnType serviceRequestResult = transmissionServer.request(pdu, transmitEventType);
165  // Test outcomes.
166  if (serviceRequestResult == FacilityReturnType::TOKEN_PUT_IN_SERVICE) {
167  scheduler.schedule(Event(transmissionTime, endTransmitEventType, pdu)); // Successful; schedule next event to release/propagate.
168  return LinkReturnType::PDU_IN_TRANSMISSION_NEXT_EVENT_SCHEDULED;
169  } else if (serviceRequestResult == FacilityReturnType::FACILITY_BUSY_QUEUE_FULL_TOKEN_DROPPED) { // Facility queue full, PDU was dropped; update statistics.
171  return LinkReturnType::LINK_BUSY_QUEUE_FULL_PDU_DROPPED;
172  } else if (serviceRequestResult == FacilityReturnType::FACILITY_DOWN_TOKEN_DROPPED) { // Facility down (thus link is also down); PDU dropped, update stats of Node?
174  return LinkReturnType::LINK_DOWN_PDU_DROPPED;
175  } else if (serviceRequestResult == FacilityReturnType::FACILITY_BUSY_TOKEN_ENQUEUED) { // Facility busy, token enqueued. Nothing else to do.
176  return LinkReturnType::LINK_BUSY_PDU_ENQUEUED;
177  } else {
178  return LinkReturnType::NOT_IMPLEMENTED; // Something weird happened...
179  }
181 }

198 LinkReturnType Link::propagatePDU(EventType nextEvent, std::shared_ptr<const ProtocolDataUnit> pdu) {
199  // Release PDU from transmission Facility.
200  transmissionServer.release(pdu);
201  // If link is down, discard PDU, update statistics.
202  if (!isUp()) {
203  ++droppedPdusCountMedium;
204  return LinkReturnType::LINK_DOWN_PDU_DROPPED;
205  }
206  // Otherwise, insert PDU into inTransitQueue and schedule next arrival event.
207  inTransitQueue.push_back(pdu); // Insert into queue.
208  scheduler.schedule(Event(propagationDelay, nextEvent, pdu));
209  return LinkReturnType::PDU_IN_TRANSIT_NEXT_EVENT_SCHEDULED;
210 }

335 unsigned int Link::purgeQueue() {
336  // Drop each PDU in inTransitQueue and update statistics.
337  unsigned int droppedPdusHere = inTransitQueue.size(); // PDUs dropped from link medium, in transit queue.
338  while (inTransitQueue.size() > 0) {
339  scheduler.removeEvents(inTransitQueue.front()); // Delete events related to first PDU.
340  inTransitQueue.pop_front(); // Remove first PDU.
341  }
342  droppedPdusCountMedium += droppedPdusHere; // Update member counter for dropped PDUs in link medium.
343  return droppedPdusHere;
344 }
